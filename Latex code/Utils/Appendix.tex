\begin{proof}[Proof of \Cref{sixseven}]
    We will prove the LHS inequality, the RHS can be proven in the exact same way : \begin{align*}
        \int a(x) d\mu(x) &= \int \frac{a(x)}{b(x)} b(x) d\mu(x) \\
        &\geq \Bigl(\inf_{x \in spt(\mu)} \frac{a(x)}{b(x)}\Bigr) \int b(x) d\mu(x)
    \end{align*}
    The last inequality is implied by the fact that $\forall x\in \mx : b(x) \geq 0$.
\end{proof}
\begin{proof}[Proof of \Cref{gaussianexpec}]
    Let $Z'$ be a standard Gaussian independent of $Z$, then : 
    \begin{equation*}
        \E\Phi(aZ+b) = \P(aZ+b\geq Z') = \P \Bigl(\frac{Z'-aZ}{\sqrt{1+a^2}} \leq \frac{b}{\sqrt{1+a^2}}\Bigr) = \Phi(\frac{b}{\sqrt{1+a^2}})
    \end{equation*}
    The last equality is given by the fact that $\frac{Z'-aZ}{\sqrt{1+a^2}} \sim \mathcal{N}(0,1)$.
\end{proof}
